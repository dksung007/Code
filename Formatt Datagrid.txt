 Private Sub FormatDgv()
        Dim cs As New DataGridViewCellStyle
        cs.Font = New Font("Ms Sans Serif", 10, FontStyle.Bold)
        With DataGridView1
            If .RowCount > 0 Then
                .ColumnHeadersDefaultCellStyle = cs
                .ReadOnly = True
                If Ck_ShowPss.Checked = False Then
                    '.Columns(4).Visible = False
                ElseIf Ck_ShowPss.Checked = True Then
                    '.Columns(4).Visible = True
                End If

                '.Columns(1).HeaderText = "รหัสพนักงาน"
                '.Columns(1).HeaderText = "ชื่อ"
                '.Columns(2).HeaderText = Split(readLine, ";")(2)
                '.Columns(3).HeaderText = "ชื่อเข้าใช้"
                '.Columns(4).HeaderText = "เแฟกซ์"
                '.Columns(5).HeaderText = "เลขประจำตัวผู้เสียภาษี"
                '.Columns(8).Visible = False
                '.Columns(9).Visible = False
                .Columns(0).Width = 200
                '.Columns(1).Width = 150
                '.Columns(2).Width = 350
                '.Columns(3).Width = 100
                '.Columns(4).Width = 100
                '.Columns(5).Width = 150

                .AlternatingRowsDefaultCellStyle.BackColor = Color.LightGreen
                .BackgroundColor = Color.White
            End If
        End With
